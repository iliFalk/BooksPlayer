document.addEventListener('DOMContentLoaded', function() {

// Telegram WebApp initialization

const tgWebApp = window.Telegram.WebApp;

// Expand the WebApp to full viewport

tgWebApp.expand();

// Set background color to match Telegram theme

document.body.style.setProperty('--tg-theme-bg-color', tgWebApp.themeParams.bg_color || '#ffffff');

document.body.style.setProperty('--tg-theme-text-color', tgWebApp.themeParams.text_color || '#000000');

document.body.style.setProperty('--tg-theme-hint-color', tgWebApp.themeParams.hint_color || '#707579');

document.body.style.setProperty('--tg-theme-link-color', tgWebApp.themeParams.link_color || '#3390ec');

// Audio player elements

const audioPlayer = new Audio();

const playBtn = document.getElementById('playBtn');

const skipBackBtn = document.getElementById('skipBackBtn');

const skipForwardBtn = document.getElementById('skipForwardBtn');

const progressBar = document.getElementById('progressBar');

const progressFill = document.getElementById('progressFill');

const currentTimeEl = document.getElementById('currentTime');

const durationEl = document.getElementById('duration');

const volumeSlider = document.getElementById('volumeSlider');

const chapterList = document.getElementById('chapterList');

const bookTitle = document.getElementById('bookTitle');

const bookCover = document.getElementById('bookCover');

const uploadContainer = document.getElementById('uploadContainer');

const playerContainer = document.getElementById('playerContainer');

const aiLoading = document.getElementById('aiLoading');

// Mock chapters data (would be generated by AI in a real app)

let chapters = [];

let currentBook = null;

// Player state

let isPlaying = false;

let isDragging = false;

// Format time (seconds to MM:SS)

function formatTime(seconds) {

const minutes = Math.floor(seconds / 60);

const remainingSeconds = Math.floor(seconds % 60);

return `${minutes}:${remainingSeconds < 10 ? '0' : ''}${remainingSeconds}`;

}

// Update progress bar

function updateProgress() {

if (!isDragging) {

const currentTime = audioPlayer.currentTime;

const duration = audioPlayer.duration || 1;

const percentage = (currentTime / duration) * 100;

progressFill.style.width = `${percentage}%`;

currentTimeEl.textContent = formatTime(currentTime);

// Highlight current chapter

if (chapters.length > 0) {

const chapterItems = document.querySelectorAll('.chapter-item');

chapterItems.forEach((item, index) => {

const chapter = chapters[index];

if (currentTime >= chapter.startTime &&

(index === chapters.length - 1 || currentTime < chapters[index + 1].startTime)) {

item.classList.add('active');

} else {

item.classList.remove('active');

}

});

}

}

}

// Set audio duration when metadata is loaded

audioPlayer.addEventListener('loadedmetadata', function() {

durationEl.textContent = formatTime(audioPlayer.duration);

});

// Update progress while playing

audioPlayer.addEventListener('timeupdate', updateProgress);

// Play/pause toggle

function togglePlay() {

if (currentBook) {

if (isPlaying) {

audioPlayer.pause();

playBtn.innerHTML = '<i class="fas fa-play"></i>';

} else {

audioPlayer.play()

.then(() => {

playBtn.innerHTML = '<i class="fas fa-pause"></i>';

})

.catch(error => {

console.error('Playback failed:', error);

// Show error to user

tgWebApp.showAlert('Playback failed. Please try again.');

});

}

isPlaying = !isPlaying;

}

}

playBtn.addEventListener('click', togglePlay);

// Skip forward/backward

skipBackBtn.addEventListener('click', function() {

if (currentBook) {

audioPlayer.currentTime = Math.max(0, audioPlayer.currentTime - 15);

}

});

skipForwardBtn.addEventListener('click', function() {

if (currentBook) {

audioPlayer.currentTime = Math.min(audioPlayer.duration, audioPlayer.currentTime + 15);

}

});

// Handle progress bar scrubbing

progressBar.addEventListener('mousedown', function(e) {

if (currentBook) {

isDragging = true;

updateProgressOnClick(e);

}

});

document.addEventListener('mousemove', function(e) {

if (isDragging && currentBook) {

updateProgressOnClick(e);

}

});

document.addEventListener('mouseup', function() {

isDragging = false;

});

function updateProgressOnClick(e) {

const rect = progressBar.getBoundingClientRect();

const pos = (e.clientX - rect.left) / rect.width;

const newTime = pos * audioPlayer.duration;

audioPlayer.currentTime = newTime;

progressFill.style.width = `${pos * 100}%`;

currentTimeEl.textContent = formatTime(newTime);

}

// Handle volume control

volumeSlider.addEventListener('input', function() {

audioPlayer.volume = this.value / 100;

});

// File upload/forward handling

uploadContainer.addEventListener('click', function() {

// In a real Telegram WebApp, we would use the file picker

// For demo purposes, we'll simulate a file selection

const demoAudioUrls = [

{

url: 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3',

title: 'Sample Audiobook',

coverColor: '#6e8efb'

},

{

url: 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-2.mp3',

title: 'Classic Literature',

coverColor: '#a777e3'

}

];

// Show a simple alert with "options"

tgWebApp.showPopup({

title: 'Select Demo Audio',

message: 'Choose a sample audiobook to test the player:',

buttons: [

{id: 'book1', type: 'default', text: 'Sample Audiobook'},

{id: 'book2', type: 'default', text: 'Classic Literature'},

{id: 'cancel', type: 'cancel'}

]

}, function(buttonId) {

if (buttonId === 'book1') {

loadBook(demoAudioUrls[0]);

} else if (buttonId === 'book2') {

loadBook(demoAudioUrls[1]);

}

});

});

// Load a book into the player

function loadBook(bookData) {

currentBook = bookData;

audioPlayer.src = bookData.url;

bookTitle.textContent = bookData.title;

bookCover.style.background = `linear-gradient(135deg, ${bookData.coverColor}, ${darkenColor(bookData.coverColor, 20)})`;

// Enable player controls

playBtn.disabled = false;

// Show loading indicator for AI processing

aiLoading.style.display = 'inline-block';

// In a real app, we would send the audio to an AI service for analysis

// For demo, we'll simulate this with a timeout and mock data

setTimeout(generateMockChapters, 2000);

}

// Generate mock chapters (simulating AI processing)

function generateMockChapters() {

// Hide loading indicator

aiLoading.style.display = 'none';

// Create mock chapters based on the book title

const chapterTitles = [

"Introduction",

"The Beginning",

"Rising Action",

"Climax",

"Falling Action",

"Conclusion"

];

if (currentBook.title.includes("Classic")) {

chapterTitles[1] = "Chapter I: The Early Years";

chapterTitles[2] = "Chapter II: The Journey Begins";

chapterTitles[3] = "Chapter III: The Great Adventure";

chapterTitles[4] = "Chapter IV: The Final Confrontation";

chapterTitles[5] = "Epilogue";

}

chapters = [];

const duration = 1800; // Assuming 30 minute duration for demo

const chapterDuration = duration / chapterTitles.length;

for (let i = 0; i < chapterTitles.length; i++) {

chapters.push({

title: chapterTitles[i],

description: `Key events and narrative development in ${chapterTitles[i].toLowerCase()}`,

startTime: i * chapterDuration,

duration: chapterDuration

});

}

// Display chapters

renderChapters();

// Show success message

tgWebApp.showAlert(`AI Analysis Complete: ${chapters.length} chapters detected`);

}

// Render chapters list

function renderChapters() {

chapterList.innerHTML = '';

chapters.forEach((chapter, index) => {

const chapterItem = document.createElement('div');

chapterItem.className = 'chapter-item';

chapterItem.innerHTML = `

<div class="chapter-number">${index + 1}</div>

<div class="chapter-content">

<div class="chapter-title">${chapter.title}</div>

<div class="chapter-time">${formatTime(chapter.startTime)}</div>

</div>

`;

// Navigate to chapter on click

chapterItem.addEventListener('click', function() {

audioPlayer.currentTime = chapter.startTime;

if (!isPlaying) {

togglePlay();

}

// Scroll the chapter into view with offset

setTimeout(() => {

chapterItem.scrollIntoView({

behavior: 'smooth',

block: 'center'

});

}, 100);

});

chapterList.appendChild(chapterItem);

});

}

// Helper function to darken a color

function darkenColor(color, percent) {

const num = parseInt(color.replace("#", ""), 16),

amt = Math.round(2.55 * percent),

R = (num >> 16) - amt,

G = (num >> 8 & 0x00FF) - amt,

B = (num & 0x0000FF) - amt;

return '#' + (

0x1000000 +

(R < 255 ? R < 1 ? 0 : R : 255) * 0x10000 +

(G < 255 ? G < 1 ? 0 : G : 255) * 0x100 +

(B < 255 ? B < 1 ? 0 : B : 255)

).toString(16).slice(1);

}
